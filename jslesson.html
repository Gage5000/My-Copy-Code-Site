One of the most useful things to add is a copy to make copying text or code eseyer to do.


function copyIframeContent() {

const copyAt = document.getElementById('copyToText');

const copyDocument = copyAt.contentDocument || copyAt.contentWindow.document;

const textToCopy = copyDocument.body.innerText || copyDocument.body.textContent;

if (navigator.clipboard) {

navigator.clipboard.writeText(textToCopy).then(() => {

}).catch(err => {

console.error('Failed to copy: ', err);

});

} else {

// Fallback for older browsers

const textarea = document.createElement('textarea');

textarea.value = textToCopy;

document.body.appendChild(textarea);

textarea.select();

document.execCommand('copy');

document.body.removeChild(textarea);

}

}
